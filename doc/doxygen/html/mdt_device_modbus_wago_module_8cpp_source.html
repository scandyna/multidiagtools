433 bool mdtDeviceModbusWagoModule::readRegisters(int firstChannel, int lastChannel, bool needDeviceSetupState)
434 {
435  Q_ASSERT(pvDevice != 0);
436  Q_ASSERT(firstChannel >= 0);
437  Q_ASSERT(lastChannel >= 0);
438  Q_ASSERT(lastChannel < pvStatusBytes.size());
439  Q_ASSERT(lastChannel < pvRegisterDataWords.size());
440  Q_ASSERT(firstChannel <= lastChannel);
441 
442  int startAddress;
443  int registersCount;
444  int channel, i;
445 
450  // Read registers from device
451  startAddress = pvStatusBytes.at(firstChannel).first;
452  registersCount = pvStatusBytes.at(lastChannel).first + 1 - startAddress + 1;
454  qDebug() << "Module - readRegisters(): startAddress: " << startAddress << " , qty: " << registersCount;
455  if(!pvDevice->getRegisterValues(startAddress, registersCount)){
456  mdtError e(MDT_DEVICE_ERROR, "Device: " + pvDevice->name() + " : unable to get register values from device", mdtError::Error);
457  MDT_ERROR_SET_SRC(e, "mdtDeviceModbusWagoModule");
458  e.commit();
459  return false;
460  }
461  qDebug() << "Module - readRegisters(): getRegisterValues() done - values [hex]: " << hex << pvDevice->registerValues();
462  if(pvDevice->registerValues().size() != registersCount){
463  mdtError e(MDT_DEVICE_ERROR, "Device: " + pvDevice->name() + " : received unpexpected count of register values from device", mdtError::Error);
464  MDT_ERROR_SET_SRC(e, "mdtDeviceModbusWagoModule");
465  e.commit();
466  return false;
467  }
468  // Update cache
469  channel = firstChannel;
470  for(i = 0; i < pvDevice->registerValues().size(); i += 2){
471  Q_ASSERT((i + 1) < pvDevice->registerValues().size());
472  Q_ASSERT(channel < pvStatusBytes.size());
473  Q_ASSERT(channel < pvRegisterDataWords.size());
474  pvStatusBytes[channel].second = pvDevice->registerValues().at(i) & 0x00FF;
476  pvRegisterDataWords[channel] = pvDevice->registerValues().at(i + 1);
477  ++channel;
478  }
486  qDebug() << "Module - readRegisters(): status bytes [hex] : " << hex << pvStatusBytes;
487  qDebug() << "Module - readRegisters(): register data [hex]: " << hex << pvRegisterDataWords;
488 
489  return true;
490 }
