00104 {
00105     if (!isOpen()) {
00106         qWarning("QtLockedFile::lock(): file is not opened");
00107         return false;
00108     }
00109 
00110     if (mode == NoLock)
00111         return unlock();
00112 
00113     if (mode == m_lock_mode)
00114         return true;
00115 
00116     if (m_lock_mode != NoLock)
00117         unlock();
00118 
00119     if (!wmutex && !(wmutex = getMutexHandle(-1, true)))
00120         return false;
00121 
00122     if (!waitMutex(wmutex, block))
00123         return false;
00124 
00125     if (mode == ReadLock) {
00126         int idx = 0;
00127         for (; idx < MAX_READERS; idx++) {
00128             rmutex = getMutexHandle(idx, false);
00129             if (!rmutex || waitMutex(rmutex, false))
00130                 break;
00131             CloseHandle(rmutex);
00132         }
00133         bool ok = true;
00134         if (idx >= MAX_READERS) {
00135             qWarning("QtLockedFile::lock(): too many readers");
00136             rmutex = 0;
00137             ok = false;
00138         }
00139         else if (!rmutex) {
00140             rmutex = getMutexHandle(idx, true);
00141             if (!rmutex || !waitMutex(rmutex, false))
00142                 ok = false;
00143         }
00144         if (!ok && rmutex) {
00145             CloseHandle(rmutex);
00146             rmutex = 0;
00147         }
00148         ReleaseMutex(wmutex);
00149         if (!ok)
00150             return false;
00151     }
00152     else {
00153         Q_ASSERT(rmutexes.isEmpty());
00154         for (int i = 0; i < MAX_READERS; i++) {
00155             Qt::HANDLE mutex = getMutexHandle(i, false);
00156             if (mutex)
00157                 rmutexes.append(mutex);
00158         }
00159         if (rmutexes.size()) {
00160             DWORD res = WaitForMultipleObjects(rmutexes.size(), rmutexes.constData(),
00161                                                TRUE, block ? INFINITE : 0);
00162             if (res != WAIT_OBJECT_0 && res != WAIT_ABANDONED) {
00163                 if (res != WAIT_TIMEOUT)
00164                     qErrnoWarning("QtLockedFile::lock(): WaitForMultipleObjects failed");
00165                 m_lock_mode = WriteLock;  // trick unlock() to clean up - semiyucky
00166                 unlock();
00167                 return false;
00168             }
00169         }
00170     }
00171 
00172     m_lock_mode = mode;
00173     return true;
