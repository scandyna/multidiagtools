103 bool QtLockedFile::lock(LockMode mode, bool block)
104 {
105  if (!isOpen()) {
106  qWarning("QtLockedFile::lock(): file is not opened");
107  return false;
108  }
109 
110  if (mode == NoLock)
111  return unlock();
112 
113  if (mode == m_lock_mode)
114  return true;
115 
116  if (m_lock_mode != NoLock)
117  unlock();
118 
119  if (!wmutex && !(wmutex = getMutexHandle(-1, true)))
120  return false;
121 
122  if (!waitMutex(wmutex, block))
123  return false;
124 
125  if (mode == ReadLock) {
126  int idx = 0;
127  for (; idx < MAX_READERS; idx++) {
128  rmutex = getMutexHandle(idx, false);
129  if (!rmutex || waitMutex(rmutex, false))
130  break;
131  CloseHandle(rmutex);
132  }
133  bool ok = true;
134  if (idx >= MAX_READERS) {
135  qWarning("QtLockedFile::lock(): too many readers");
136  rmutex = 0;
137  ok = false;
138  }
139  else if (!rmutex) {
140  rmutex = getMutexHandle(idx, true);
141  if (!rmutex || !waitMutex(rmutex, false))
142  ok = false;
143  }
144  if (!ok && rmutex) {
145  CloseHandle(rmutex);
146  rmutex = 0;
147  }
148  ReleaseMutex(wmutex);
149  if (!ok)
150  return false;
151  }
152  else {
153  Q_ASSERT(rmutexes.isEmpty());
154  for (int i = 0; i < MAX_READERS; i++) {
155  Qt::HANDLE mutex = getMutexHandle(i, false);
156  if (mutex)
157  rmutexes.append(mutex);
158  }
159  if (rmutexes.size()) {
160  DWORD res = WaitForMultipleObjects(rmutexes.size(), rmutexes.constData(),
161  TRUE, block ? INFINITE : 0);
162  if (res != WAIT_OBJECT_0 && res != WAIT_ABANDONED) {
163  if (res != WAIT_TIMEOUT)
164  qErrnoWarning("QtLockedFile::lock(): WaitForMultipleObjects failed");
165  m_lock_mode = WriteLock; // trick unlock() to clean up - semiyucky
166  unlock();
167  return false;
168  }
169  }
170  }
171 
172  m_lock_mode = mode;
173  return true;
174 }

177 {

180  return false;
181  }
182

191  else {

194  CloseHandle(mutex);

198  }
199

201  return true;
202 }
