00200 bool mdtUsbtmcPortSetupDialog::applySetup()
00201 {
00202   mdtUsbtmcPortManager *usbtmcPortManager;
00203   QString idn;
00204   QStringList idnItems;
00205 
00207   // We must close the port
00208   portManager()->closePort();
00209   // Apply current configuration
00210   updateConfig();
00211   portManager()->setPortInfo(pvPortInfoCbHandler.currentPortInfo());
00212   // Try to open port and start port manager
00213   if(!portManager()->openPort()){
00214     showStatusMessage(tr("Cannot open port"), 3000);
00215     return false;
00216   }
00217   if(!portManager()->start()){
00218     showStatusMessage(tr("Cannot start thread"), 3000);
00219     return false;
00220   }
00221   // Get identification
00222   usbtmcPortManager = dynamic_cast<mdtUsbtmcPortManager*>(portManager());
00223   Q_ASSERT(usbtmcPortManager != 0);
00224   idn = usbtmcPortManager->sendQuery("*IDN?\n");
00225   if(idn.isEmpty()){
00226     usbtmcPortManager->closePort();
00227     showStatusMessage(tr("Cannot get device infos (*IDN? query failed)"), 3000);
00228     return false;
00229   }
00230   idn = idn.trimmed();
00231   idnItems = idn.split(',');
00232   if(idnItems.size() != 4){
00233     usbtmcPortManager->closePort();
00234     showStatusMessage(tr("Cannot get device infos (*IDN? query returned unexpected data)"), 3000);
00235     return false;
00236   }
00237   // Display infos
00238   lbManufacturer->setText(idnItems.at(0));
00239   lbModel->setText(idnItems.at(1));
00240   lbSerial->setText(idnItems.at(2));
00241   lbFirmware->setText(idnItems.at(3));
00242 
00243   return true;
00244 }
