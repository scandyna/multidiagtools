00345 mdtFrame *mdtAbstractPort::getFrameToWrite()
00346 {
00347   mdtFrame *frame;
00348 
00349   // If queue is empty, wait
00350   if(pvWriteFrames.size() < 1){
00351     pvWriteFrameAvailable.wait(&pvMutex);
00352     // During wait it can happen that a flush was requested,
00353     //  witch clears the write queue. If this happens,
00354     //  we must clear the pvFlushOutRequestPending flag,
00355     //  else thread will not write new frame that must be sent.
00356     pvFlushOutRequestPending = false;
00357   }
00358   // If abortFrameToWriteWait() was called, it can happen that queue is empty
00359   if(pvWriteFrames.size() < 1){
00360     return 0;
00361   }
00362   frame = pvWriteFrames.dequeue();
00363   Q_ASSERT(frame != 0);
00364 
00365   return frame;
