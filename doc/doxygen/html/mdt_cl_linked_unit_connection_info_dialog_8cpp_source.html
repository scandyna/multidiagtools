171 QString mdtClLinkedUnitConnectionInfoDialog::sqlForLinkedConnections(const QList<QVariant> & connectionIdList) const
172 {
173  QLocale locale;
174  QString unitFunctionField;
175  QString articleFunctionField;
176  QString sql;
177  int i;
178 
179  if(connectionIdList.isEmpty()){
180  return sql;
181  }
182  // Setup base of query
183  sql = "SELECT\n"\
184  " Unit_tbl.Cabinet,\n"\
185  " Unit_tbl.Coordinate,\n"\
186  " Unit_tbl.SchemaPosition,\n"\
187  " Unit_tbl.Alias,\n"\
188  " UnitConnector_tbl.Name AS UnitConnectorName ,\n"\
189  " UnitConnection_tbl.UnitContactName ,\n"\
190  " UnitConnection_tbl.SchemaPage ,\n"\
191  " UnitConnection_tbl.FunctionEN AS UnitFunctionEN,\n"\
192  " UnitConnection_tbl.FunctionFR AS UnitFunctionFR,\n"\
193  " UnitConnection_tbl.FunctionDE AS UnitFunctionDE,\n"\
194  " UnitConnection_tbl.FunctionIT AS UnitFunctionIT,\n"\
195  " UnitConnection_tbl.SignalName ,\n"\
196  " UnitConnection_tbl.SwAddress ,\n"\
197  " ArticleConnector_tbl.Name AS ArticleConnectorName ,\n"\
198  " ArticleConnection_tbl.ArticleContactName ,\n"\
199  " ArticleConnection_tbl.IoType ,\n"\
200  " ArticleConnection_tbl.FunctionEN AS ArticleFunctionEN,\n"\
201  " ArticleConnection_tbl.FunctionFR AS ArticleFunctionFR,\n"\
202  " ArticleConnection_tbl.FunctionDE AS ArticleFunctionDE,\n"\
203  " ArticleConnection_tbl.FunctionIT AS ArticleFunctionIT\n"\
204  "FROM UnitConnection_tbl\n"\
205  " LEFT JOIN UnitConnector_tbl\n"\
206  " ON UnitConnector_tbl.Id_PK = UnitConnection_tbl.UnitConnector_Id_FK\n"\
207  " LEFT JOIN ArticleConnector_tbl\n"\
208  " ON UnitConnector_tbl.ArticleConnector_Id_FK = ArticleConnector_tbl.Id_PK\n"\
209  " LEFT JOIN ArticleConnection_tbl\n"\
210  " ON UnitConnection_tbl.ArticleConnection_Id_FK = ArticleConnection_tbl.Id_PK\n"\
211  " JOIN Unit_tbl\n"\
212  " ON Unit_tbl.Id_PK = UnitConnection_tbl.Unit_Id_FK";
213  // Add filter on requested connection IDs
214  Q_ASSERT(connectionIdList.size() > 0);
215  sql += " WHERE UnitConnection_tbl.Id_PK = " + connectionIdList.at(0).toString();
216  for(i = 1; i < connectionIdList.size(); ++i){
217  sql += " OR UnitConnection_tbl.Id_PK = " + connectionIdList.at(i).toString();
218  }
219 
220  return sql;
221 }
