70 void mdtPortStatusWidget::enableTxRxLeds(mdtPortThread *txThread, mdtPortThread *rxThread)

93  // Make connections
94  pvTxThread = txThread;
95  pvRxThread = rxThread;
96  connect(pvTxThread, SIGNAL(writeProcessBegin()), this, SLOT(trigTxLed()));
97  connect(pvRxThread, SIGNAL(readProcessBegin()), this, SLOT(trigRxLed()));
