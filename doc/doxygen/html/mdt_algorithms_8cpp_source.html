38 bool mdtAlgorithms::naturalCompareLessThan(const QString &str1, const QString &str2, Qt::CaseSensitivity caseSensitivity)
39 {
40  QString a;
41  QString b;
42  const QChar *currA;
43  const QChar *currB;
44  const QChar *begSeqA;
45  const QChar *begSeqB;
46  int cmp;
47  bool isFirstRun;
48  int weight;
49 
50  if(caseSensitivity == Qt::CaseSensitive){
51  a = str1;
52  b = str2;
53  }else{
54  a = str1.toLower();
55  b = str2.toLower();
56  }
57  // Get iterators over a and b
58  currA = a.unicode();
59  currB = b.unicode();
60  // If str1 (a) and str2 (b) are the same string, we have no difference
61  if(currA == currB){
62  return false;
63  }
64 
65  while((!currA->isNull()) && (!currB->isNull())){
66  // Remember beginning of sequences
67  begSeqA = currA;
68  begSeqB = currB;
69  // Check about special characters
70  if(currA->unicode() == QChar::ObjectReplacementCharacter){
71  return false;
72  }
73  if(currB->unicode() == QChar::ObjectReplacementCharacter){
74  return true;
75  }
76  if(currA->unicode() == QChar::ReplacementCharacter){
77  return false;
78  }
79  if(currB->unicode() == QChar::ReplacementCharacter){
80  return true;
81  }
82  // find sequence of characters ending at the first non-character
83  while((!currA->isNull()) && (!currA->isDigit()) && (!currA->isPunct()) && (!currA->isSpace())){
84  ++currA;
85  }
86  while((!currB->isNull()) && (!currB->isDigit()) && (!currB->isPunct()) && (!currB->isSpace())){
87  ++currB;
88  }
89  // compare these sequences
90  const QStringRef &subA(a.midRef(begSeqA - a.unicode(), currA - begSeqA));
91  const QStringRef &subB(b.midRef(begSeqB - b.unicode(), currB - begSeqB));
92  cmp = QStringRef::localeAwareCompare(subA, subB);
93  if(cmp < 0){
94  return true;
95  }else if(cmp > 0){
96  return false;
97  }
98  if(currA->isNull() || currB->isNull()){
99  break;
100  }
101  // find sequence of characters ending at the first non-character
102  while((currA->isPunct() || currA->isSpace()) && (currB->isPunct() || currB->isSpace())){
103  if(*currA < *currB){
104  return true;
105  }else if(*currA > *currB){
106  return false;
107  }
108  ++currA;
109  ++currB;
110  if(currA->isNull() || currB->isNull()){
111  break;
112  }
113  }
114  // now some digits follow...
115  if((*currA == QChar('0')) || (*currB == QChar('0'))){
116  // one digit-sequence starts with 0 -> assume we are in a fraction part
117  // do left aligned comparison (numbers are considered left aligned)
118  while(1){
119  if((!currA->isDigit()) && (!currB->isDigit())){
120  break;
121  }else if(!currA->isDigit()){
122  return false;
123  }else if(!currB->isDigit()){
124  return true;
125  }else if(*currA < *currB){
126  return true;
127  }else if(*currA > *currB){
128  return false;
129  }
130  ++currA;
131  ++currB;
132  }
133  }else{
134  // no digit-sequence starts with 0, assume we are looking at some integer
135  // do right aligned comparison
136  // The longest run of digits wins. That aside, the gratest value wins,
137  // but we can't know that it will until we have scanned both numbers
138  // to know that they have the same magnitude
139  isFirstRun = true;
140  weight = 0;
141  while(1){
142  if((!currA->isDigit()) && (!currB->isDigit())){
143  if(weight < 0){
144  return true;
145  }else if(weight > 0){
146  return false;
147  }
148  break;
149  }else if(!currA->isDigit()){
150  if(isFirstRun){
151  if(*currA < *currB){
152  return true;
153  }else{
154  return false;
155  }
156  }else{
157  return true;
158  }
159  }else if(!currB->isDigit()){
160  if(isFirstRun){
161  if(*currA < *currB){
162  return true;
163  }else{
164  return false;
165  }
166  }else{
167  return false;
168  }
169  }else if((*currA < *currB) && (weight == 0)){
170  weight = -1;
171  }else if((*currA > *currB) && (weight == 0)){
172  weight = +1;
173  }
174  ++currA;
175  ++currB;
176  isFirstRun = false;
177  }
178  }
179  }
180  if(currA->isNull() && currB->isNull()){
181  return false;
182  }
183  if(currA->isNull()){
184  return true;
185  }
186  return false;
187 }

246 QString mdtAlgorithms::unprotectedString(const QString &str, const QString &dataProtection, const QChar &escapeChar, int *strEndOffset, int strStartOffset)
247 {
248  QString result;
249  int dpCursor = -1;
250  int dpCursorMem = -1;
251  int remainingDp = 2;
252  int start;
253  int end = 0;
254  bool escapeEqualProtection = QString(escapeChar) == dataProtection;
255 
256  if(strStartOffset > -1){
257  start = strStartOffset;
258  }else{
259  start = 0;
260  }
261 
262  while((end < str.size())&&(remainingDp > 0)){
263  // Get index of data protection
264  if(!dataProtection.isEmpty()){
265  dpCursor = str.indexOf(dataProtection, start);
266  }else{
267  dpCursor = -1;
268  }
269  if(dpCursor < 0){
270  // Finish parsing
271  dpCursor = dpCursorMem;
272  break;
273  }else{
274  if(escapeEqualProtection){
275  if(dpCursor < (str.size()-1)){
276  // check about escaped sequence
277  if(str.at(dpCursor+1) == escapeChar){
278  // Escaped sequence
279  end = dpCursor + 1;
280  result.append(str.mid(start, end-start));
281  start = dpCursor + dataProtection.size() + 1;
282  }else{
283  --remainingDp;
284  end = dpCursor;
285  result.append(str.mid(start, end-start));
286  start = dpCursor + dataProtection.size();
287  }
288  }else{
289  end = dpCursor;
290  result.append(str.mid(start, end-start));
291  break;
292  }
293  }else{
294  if(dpCursor == 0){
295  // Just move start cursor after protection
296  start = dpCursor + dataProtection.size();
297  --remainingDp;
298  }else{
299  if(str.at(dpCursor-1) == escapeChar){
300  // Escaped sequence
301  end = dpCursor-1;
302  result.append(str.mid(start, end-start));
303  // Add escape sequence
304  result.append(str.mid(dpCursor, dataProtection.size()));
305  start = dpCursor + dataProtection.size();
306  }else{
307  --remainingDp;
308  end = dpCursor;
309  result.append(str.mid(start, end-start));
310  start = dpCursor + dataProtection.size();
311  }
312  }
313  }
314  }
315  dpCursorMem = dpCursor;
316  end += dataProtection.size();
317  }
318  if(strEndOffset != 0){
319  *strEndOffset = dpCursor;
320  }
321 
322  return result;
323 }

325 QStringList mdtAlgorithms::splitString(const QString &str, const QString &separator, const QString &dataProtection, const QChar &escapeChar)
326 {
327  int sepCursor = 0; // Cursor in str string for separator
328  int dpCursor = 0; // Cursor in str string for dataProtection
329  QStringList fields;
330  int start = 0;
331  int end = 0;
332 
333  while(end < str.size()){
334  // Get separator and data protection indexes
335  if(!separator.isEmpty()){
336  sepCursor = str.indexOf(separator, start);
337  }else{
338  sepCursor = -1;
339  }
340  if(!dataProtection.isEmpty()){
341  dpCursor = str.indexOf(dataProtection, start);
342  }else{
343  dpCursor = -1;
344  }
345  if((sepCursor < 0)&&(dpCursor < 0)){
346  // No keyword found, copy remaining string
347  end = str.size();
348  if(start != end){
349  fields.append(str.mid(start, end-start));
350  }
351  }else if((sepCursor < 0)&&(dpCursor > -1)){
352  // Only data protection was found, copy until next data protection (or until end if not found)
353  fields.append(unprotectedString(str, dataProtection, escapeChar, &end, start));
354  // Check about end of string
355  if(end < 0){
356  break;
357  }
358  end += dataProtection.size();
359  start = end;
360  }else if((sepCursor > -1)&&(dpCursor < 0)){
361  // Only separator was found, copy left part
362  end = sepCursor;
363  fields.append(str.mid(start, end-start));
364  start = end + separator.size();
365  }else{
366  // Separator and data protection found
367  if(dpCursor < sepCursor){
368  // Data protection was first found, copy until next data protection (or until end if not found)
369  fields.append(unprotectedString(str, dataProtection, escapeChar, &end, start));
370  // Check about end of string
371  if(end < 0){
372  break;
373  }
374  end += dataProtection.size();
375  start = end + separator.size();
376  }else{
377  // Separator was first found, copy left part
378  end = sepCursor;
379  fields.append(str.mid(start, end-start));
380  start = end + separator.size();
381  }
382  }
383  }
384  // Append a field if str terminates with a separator
385  if((!separator.isEmpty())&&(str.indexOf(separator, end-separator.size())>-1)){
386  fields.append("");
387  }
388  return fields;
389 }
