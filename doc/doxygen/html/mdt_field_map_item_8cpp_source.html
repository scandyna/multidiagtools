00158 QVariant mdtFieldMapItem::destinationData(const QString &sourceData) const
00159 {
00160   QVariant data;
00161 
00162   if((pvSourceFieldDataStartOffset < 0)&&(pvSourceFieldDataEndOffset < 0)){
00163     data = sourceData;
00164     data.convert(pvDestinationField.sqlField().type());
00165     return data;
00166   }
00167   if(pvSourceFieldDataStartOffset < 0){
00168     data = sourceData.left(pvSourceFieldDataEndOffset + 1);
00169     data.convert(pvDestinationField.sqlField().type());
00170     return data;
00171   }
00172   if(pvSourceFieldDataEndOffset < 0){
00173     data = sourceData.mid(pvSourceFieldDataStartOffset, -1);
00174     data.convert(pvDestinationField.sqlField().type());
00175     return data;
00176   }
00177   data = sourceData.mid(pvSourceFieldDataStartOffset, pvSourceFieldDataEndOffset - pvSourceFieldDataStartOffset + 1);
00178   data.convert(pvDestinationField.sqlField().type());
00179
