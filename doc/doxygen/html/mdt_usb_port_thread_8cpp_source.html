84  }

86  if(portError != mdtAbstractPort::NoError){

102  usbPort()->unlockMutex();
103  msleep(50);
104  usbPort()->lockMutex();
105  }
106  }else{
107  // Here, we hope that caller knows what hi is doing - simply restore frame to pool
108  mdtError e(MDT_USB_IO_ERROR, "Loosing a control transfer response frame", mdtError::Warning);
109  MDT_ERROR_SET_SRC(e, "mdtUsbPortThread");
110  e.commit();
111  usbPort()->controlFramesPool().enqueue(usbPort()->controlResponseFrames().dequeue());
112  }
113  }
114  Q_ASSERT(usbPort()->controlFramesPool().size() > 0);
115

206 void mdtUsbPortThread::run()
207 {
208  Q_ASSERT(pvPort != 0);
209 
210  mdtAbstractPort::error_t portError = mdtAbstractPort::NoError;
211  int n;
212  int i;
213  mdtUsbPort *port;
214  bool waitAnAnswer = false;
215  qint64 written;
216  QList<mdtAbstractPort::error_t> errors;
217  mdtPortThreadHelperPort threadHelper;
218  threadHelper.setPort(pvPort);
219  threadHelper.setThread(this);
220 
221  pvPort->lockMutex();
222 #ifdef Q_OS_UNIX
223  pvNativePthreadObject = pthread_self();
224  Q_ASSERT(pvNativePthreadObject != 0);
225 #endif
226  // We need a mdtUsbPort object here
228  port = usbPort();
229  Q_ASSERT(port != 0);
230  // Set the running flag
231  pvRunning = true;
232  // Get a frame for read
233  if(!threadHelper.getNewFrameRead()){
234  pvRunning = false;
235  notifyError(mdtAbstractPort::Disconnected);
236  return;
237  }
238  // Notify that we are ready
240  emit ready(this);
241 
242  // Run...
243  while(1){
244  // Read thread state
245  if(!pvRunning){
246  break;
247  }
248  portError = port->handleUsbEvents();
249  // Check about event handling errors
250  if(portError != mdtAbstractPort::NoError){
251  // Check about stoping
252  if(!pvRunning){
253  break;
254  }
255  portError = threadHelper.handleCommonReadWriteErrors(portError);
256  if(portError != mdtAbstractPort::ErrorHandled){
257  // Unhandled error - stop
258  break;
259  }
260  // Error could be handled - go back IDLE
261  continue;
262  }
263  // Check about port errors
264  errors = port->lastErrors();
265  portError = mdtAbstractPort::NoError;
266  for(i=0; i<errors.size(); ++i){
267  // Check about stoping
268  if(!pvRunning){
269  break;
270  }
271  portError = errors.at(i);
272  if((portError == mdtAbstractPort::ControlCanceled)||(portError == mdtAbstractPort::ControlTimeout)){
273  // mdtUsbPort does the job, we just have to notify the error
274  notifyError(mdtAbstractPort::ControlCanceled);
275  }else{
276  portError = threadHelper.handleCommonReadWriteErrors(portError);
277  if(portError != mdtAbstractPort::ErrorHandled){
278  // Unhandled error - stop
279  pvRunning = false;
280  break;
281  }
282  }
283  }
284  // Check about stoping
285  if(!pvRunning){
286  break;
287  }
288  // If we had a error, go back IDLE
289  if(portError == mdtAbstractPort::ErrorHandled){
290  continue;
291  }
292  // Check if a control response is available
293  if(port->controlResponseFrames().size() > 0){
294  emit(controlResponseReaden());
295  }
296  // Check if a message IN is available
297  if(port->messageInFrames().size() > 0){
298  emit(messageInReaden());
299  }
300  // Init a new message IN transfert (will only init if needed)
301  portError = port->initMessageInTransfer();
302  if(portError != mdtAbstractPort::NoError){
303  // Check about stoping
304  if(!pvRunning){
305  break;
306  }
307  portError = threadHelper.handleCommonReadWriteErrors(portError);
308  if(portError != mdtAbstractPort::ErrorHandled){
309  // Unhandled error - stop
310  break;
311  }
312  // Error could be handled - go back IDLE
313  continue;
314  }
315  // Check about write endpoint flush
316  if((threadHelper.currentWriteFrame() != 0)&&(port->flushOutRequestPending())){
317  threadHelper.restoreCurrentWriteFrameToPool();
318  }else{
319  // Check if we have something to write (we not want to block here)
320  if(threadHelper.currentWriteFrame() == 0){
321  if(port->writeFrames().size() > 0){
322  threadHelper.setCurrentWriteFrame(port->writeFrames().dequeue());
323  Q_ASSERT(threadHelper.currentWriteFrame() != 0);
324  emit(writeProcessBegin());
325  }
326  }
327  }
328  // Write ...
329  if(threadHelper.currentWriteFrame() != 0){
330  // Write (will simply do nothing and return 0 if transfer is pending)
331  written = threadHelper.writeDataToPort(-1);
332  if(written < 0){
333  // Check about stoping
334  if(!pvRunning){
335  break;
336  }
337  portError = threadHelper.handleCommonWriteErrors((mdtAbstractPort::error_t)written);
338  if(portError != mdtAbstractPort::ErrorHandled){
339  // Unhandled error - stop
340  break;
341  }
342  // Error could be handled - go back IDLE
343  continue;
344  }
345  // Check if frame was completly written
346  if(threadHelper.currentWriteFrame()->isEmpty()){
347  // Update waitAnAnswer flag
348  waitAnAnswer = threadHelper.currentWriteFrame()->waitAnAnswer();
349  // Restore frame to pool
350  threadHelper.restoreCurrentWriteFrameToPool();
351  // Check if a new frame is to write
352  if(port->writeFrames().size() > 0){
353  threadHelper.setCurrentWriteFrame(port->writeFrames().dequeue());
354  Q_ASSERT(threadHelper.currentWriteFrame() != 0);
355  }
356  }
357  // Here, if frame is not Null, we have to init a new transfer
358  if(threadHelper.currentWriteFrame() != 0){
359  // Init a new write transfer (will only init if not pending)
360  portError = port->initWriteTransfer(threadHelper.currentWriteFrame()->data(), threadHelper.currentWriteFrame()->size());
361  if(portError != mdtAbstractPort::NoError){
362  // Check about stoping
363  if(!pvRunning){
364  break;
365  }
366  portError = threadHelper.handleCommonWriteErrors(portError);
367  if(portError != mdtAbstractPort::ErrorHandled){
368  // Unhandled error - stop
369  break;
370  }
371  // Error could be handled - go back IDLE
372  continue;
373  }
374  }
375  }
376  Q_ASSERT(threadHelper.currentReadFrame() != 0);
377  // Check about read endpoint flush
378  if(port->flushInRequestPending()){
379  threadHelper.restoreCurrentReadFrameToPool();
380  if(!threadHelper.getNewFrameRead()){
381  break;
382  }
383  // We do nothing else, return idle
384  continue;
385  }
386  // read/store available data
387  n = threadHelper.readFromPort(true);
388  if(n < 0){
389  // Unhandled error: notify and stop
390  portError = (mdtAbstractPort::error_t)n;
391  notifyError(portError);
392  break;
393  }
394  // Reset waitAnAnswer flag if a frame was received
395  if(n > 0){
396  waitAnAnswer = false;
397  }
398  // Two conditions to init a read transfer:
399  // - A query/reply is pending (waitAnAnswer)
400  // - Current read frame is not complete
401  if((waitAnAnswer)&&(n == 0)){
402  // Init a new read transfer (will only init if not pending)
403  portError = port->initReadTransfer(threadHelper.currentReadFrame()->bytesToStore());
404  if(portError != mdtAbstractPort::NoError){
405  // Check about stoping
406  if(!pvRunning){
407  break;
408  }
409  portError = threadHelper.handleCommonReadErrors(portError);
410  if(portError != mdtAbstractPort::ErrorHandled){
411  // Unhandled error - stop
412  break;
413  }
414  }
415  }
416  }
417 
418  port->cancelTransfers();
419 
420  pvRunning = false;
426  pvPort->unlockMutex();
427 }
