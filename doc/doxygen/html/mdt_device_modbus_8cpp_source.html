227  _existingHwNodeIdList.append(pvHardwareNodeId.toInt());
228  }
229  // Check that port manager is not running
230  if(!pvTcpPortManager->isClosed()){
231  pvTcpPortManager->stop();
232  }
233  // Scan looking in chache file first
234  portInfoList = pvTcpPortManager->scan(pvTcpPortManager->readScanResult(), 100, _existingHwNodeIdList, pvHardwareNodeIdBitsCount, pvHardwareNodeIdBitsStartFrom);
235  // Try to connect

237  qDeleteAll(portInfoList);
238  portInfoList.clear();
239  if(portError == mdtAbstractPort::NoError){
240  return mdtAbstractPort::NoError;
241  }
242  // Expected device not found, scan the network
243  portInfoList = pvTcpPortManager->scan(QNetworkInterface::allInterfaces(), 502, 100, true, _existingHwNodeIdList, pvHardwareNodeIdBitsCount, pvHardwareNodeIdBitsStartFrom);
244  // Try to connect

246  if(portError == mdtAbstractPort::NoError){
247  pvTcpPortManager->saveScanResult(portInfoList);
248  }
249  qDeleteAll(portInfoList);
250  portInfoList.clear();
251 
252  return portError;

276 void mdtDeviceModbus::decodeReadenFrame(mdtPortTransaction *transaction)
277 {
278  Q_ASSERT(transaction != 0);
279  if(ios() == 0){
280  return;
281  }
282 
283  int fc;
284  QVariant var;
285 
287  // Decode readen frame and update I/O's
288  fc = pvCodec->decode(transaction->data());
289  switch(fc){
290  case 0x01: // Read coils
291  // Check if we have a answer for a single I/O or multiple I/Os
292  if((transaction->ioCount() == 1)&&(transaction->digitalIo() != 0)){
293  // If one DO was requested, 1 byte is returned with just first bit set
294  if(pvCodec->values().size() == 8){
295  var = pvCodec->values().at(0);
296  if((var.isValid())&&(var.type() == QVariant::Bool)){
297  transaction->digitalIo()->setValue(var.toBool(), false);
298  }else{
299  transaction->digitalIo()->setValue(mdtValue(), false);
300  }
301  }else{
302  transaction->digitalIo()->setValue(mdtValue(), false);
303  }
304  transaction->digitalIo()->setEnabled(true);
305  }else{
306  // Transaction contains first I/O address (for read access) and qty of I/Os
307  ios()->updateDigitalOutputValues(pvCodec->values(), transaction->address(), mdtDeviceIos::Read , transaction->ioCount(), true);
308  ios()->setDigitalOutputsEnabled(true);
309  }
310  break;
311  case 0x02: // Read discrete inputs
312  // Check if we have a answer for a single I/O or multiple I/Os
313  if((transaction->ioCount() == 1)&&(transaction->digitalIo() != 0)){
314  // If one DI was requested, 1 byte is returned with just first bit set
315  if(pvCodec->values().size() == 8){
316  var = pvCodec->values().at(0);
317  if((var.isValid())&&(var.type() == QVariant::Bool)){
318  transaction->digitalIo()->setValue(var.toBool(), false);
319  }else{
320  transaction->digitalIo()->setValue(mdtValue(), false);
321  }
322  }else{
323  transaction->digitalIo()->setValue(mdtValue(), false);
324  }
325  }else{
326  // Transaction contains first I/O address and qty of I/Os
327  ios()->updateDigitalInputValues(pvCodec->values(), transaction->address(), transaction->ioCount(), true);
328  }
329  break;
330  case 0x03: // Read holding registers
331  // Check if we have a answer for a single I/O or multiple I/Os
332  if((transaction->ioCount() == 1)&&(transaction->analogIo() != 0)){
333  // Single I/O reply
334  if(pvCodec->values().size() == 1){
335  var = pvCodec->values().at(0);
336  if((var.isValid())&&(var.type() == QVariant::Int)){
337  // Codec only get raw data, so let I/O do needed conversions
338  transaction->analogIo()->setValue(var.toInt(), false);
339  }else{
340  transaction->analogIo()->setValue(mdtValue(), false);
341  }
342  }else{
343  transaction->analogIo()->setValue(mdtValue(), false);
344  }
345  transaction->analogIo()->setEnabled(true);
346  }else{
347  // Multiple I/O reply
348  ios()->updateAnalogOutputValues(pvCodec->values(), transaction->address(), mdtDeviceIos::Read, transaction->ioCount(), true);
349  ios()->setAnalogOutputsEnabled(true);
350  }
351  break;
352  case 0x04: // Read input registers
353  // Check if we have a answer for a single I/O or multiple I/Os
354  if((transaction->ioCount() == 1)&&(transaction->analogIo() != 0)){
355  // Single I/O reply
356  if(pvCodec->values().size() == 1){
357  var = pvCodec->values().at(0);
358  if((var.isValid())&&(var.type() == QVariant::Int)){
359  // Codec only get raw data, so let I/O do needed conversions
360  transaction->analogIo()->setValue(var.toInt(), false);
361  }else{
362  transaction->analogIo()->setValue(mdtValue(), false);
363  }
364  }else{
365  transaction->analogIo()->setValue(mdtValue(), false);
366  }
367  }else{
368  // Multiple I/O reply
369  ios()->updateAnalogInputValues(pvCodec->values(), transaction->address(), transaction->ioCount(), true);
370  }
371  break;
372  case 0x05: // Write single coil reply
373  if(transaction->digitalIo() != 0){
374  // Check validity and update
375  if(pvCodec->values().size() == 2){
376  // Address
377  var = pvCodec->values().at(0);
378  if((var.isValid())&&(var.type() == QVariant::Int)&&(var.toInt() == transaction->digitalIo()->addressWrite())){
379  // Value
380  var = pvCodec->values().at(1);
381  if((var.isValid())&&(var.type() == QVariant::Bool)){
382  transaction->digitalIo()->setValue(var.toBool(), false);
383  }else{
384  transaction->digitalIo()->setValue(mdtValue(), false);
385  }
386  }else{
387  transaction->digitalIo()->setValue(mdtValue(), false);
388  }
389  }else{
390  transaction->digitalIo()->setValue(mdtValue(), false);
391  }
392  transaction->digitalIo()->setEnabled(true);
393  }
394  break;
395  case 0x06: // Write single register reply
396  if(transaction->analogIo() != 0){
397  // Check validity and update
398  if(pvCodec->values().size() == 2){
399  // Address
400  var = pvCodec->values().at(0);
401  if((var.isValid())&&(var.type() == QVariant::Int)&&(var.toInt() == transaction->analogIo()->addressWrite())){
402  // Value
403  var = pvCodec->values().at(1);
404  if((var.isValid())&&(var.type() == QVariant::Int)){
405  // Update (G)UI - Codec only get raw data, so let I/O do needed conversions
406  transaction->analogIo()->setValue(var.toInt(), false);
407  }
408  }else{
409  transaction->analogIo()->setValue(mdtValue(), false);
410  }
411  }else{
412  transaction->analogIo()->setValue(mdtValue(), false);
413  }
414  transaction->analogIo()->setEnabled(true);
415  }
416  break;
417  case 0x0F: // Write multiple coils
418  // Check validitiy and update
419  if(pvCodec->values().size() == 2){
420  // Start address
421  var = pvCodec->values().at(0);
422  if((var.isValid())&&(var.type() == QVariant::Int)&&(var.toInt() == transaction->address())){
423  // Qty of outputs set
424  var = pvCodec->values().at(1);
425  if((!var.isValid())||(var.type() != QVariant::Int)||(var.toInt() != transaction->ioCount())){
426  ios()->setDigitalOutputsValue(mdtValue());
427  }
428  }else{
429  ios()->setDigitalOutputsValue(mdtValue());
430  }
431  }else{
432  ios()->setDigitalOutputsValue(mdtValue());
433  }
434  ios()->setDigitalOutputsEnabled(true);
435  case 0x10: // Write multiple registers reply
436  // Check validitiy and update
437  if(pvCodec->values().size() == 2){
438  // Start address
439  var = pvCodec->values().at(0);
440  if((var.isValid())&&(var.type() == QVariant::Int)&&(var.toInt() == transaction->address())){
441  // Qty of outputs set
442  var = pvCodec->values().at(1);
443  if((!var.isValid())||(var.type() != QVariant::Int)||(var.toInt() != transaction->ioCount())){
444  ios()->setAnalogOutputsValue(mdtValue());
445  }
446  }else{
447  ios()->setAnalogOutputsValue(mdtValue());
448  }
449  }else{
450  ios()->setAnalogOutputsValue(mdtValue());
451  }
452  ios()->setAnalogOutputsEnabled(true);
453  break;
454  default:
456  mdtError e(MDT_DEVICE_ERROR, "Received frame with unhandled function code (0x" + QString::number(fc, 16) + ")", mdtError::Warning);
457  MDT_ERROR_SET_SRC(e, "mdtDeviceModbus");
458  e.commit();
459  }
